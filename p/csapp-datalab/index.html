/* howManyBits - return the minimum number of bits required to represent x in

/* 
 * floatScale2 - Return bit-level equivalent of expression 2*f for
 *   floating point argument f.
 *   Both the argument and result are passed as unsigned int's, but
 *   they are to be interpreted as the bit-level representation of
 *   single-precision floating point values.
 *   When argument is NaN, return argument
 *   Legal ops: Any integer/unsigned operations incl. ||, &&. also if, while
 *   Max ops: 30
 *   Rating: 4
 */
unsigned floatScale2(unsigned uf) {
    unsigned s = uf & (1 << 31);
    unsigned exp = (uf & 0x7f800000) >> 23;
    unsigned frac = uf & (~0xff800000);

    if (exp == 0) return frac << 1 | s; // ÈùûËßÑÊ†ºÊï∞‰πò2Âç≥ÂèØ
    if (exp == 255) return uf;  //  ÁâπÊÆäÂÄºÁõ¥Êé•ËøîÂõû
    exp++;
    if (exp == 255) return 0x7f800000 | s;  // ‰πòÊ≥ï‰∏çÂèØËÉΩÂæóÂà∞NaNÔºåÊâÄ‰ª•ËøîÂõûÊó†Á©∑
    return s | (exp << 23) | frac;

/* 
 * floatFloat2Int - Return bit-level equivalent of expression (int) f
 *   for floating point argument f.
 *   Argument is passed as unsigned int, but
 *   it is to be interpreted as the bit-level representation of a
 *   single-precision floating point value.
 *   Anything out of range (including NaN and infinity) should return
 *   0x80000000u.
 *   Legal ops: Any integer/unsigned operations incl. ||, &&. also if, while
 *   Max ops: 30
 *   Rating: 4
 */
int floatFloat2Int(unsigned uf) {
    unsigned s = uf & (1 << 31);
    unsigned exp = (uf & 0x7f800000) >> 23;
    unsigned frac = uf & (~0xff800000);

    int E = exp - 127;
    if (exp == 255 || E > 31) return 0x80000000;  // ÁâπÊÆäÂÄº
    if (E < 0) return 0; // Â∞èÊï∞ËàçÂÖ•‰∏∫0

    unsigned M = frac | (1 << 23);  // 1 + frac
    int V = (E > 23 ? M << (E - 23) : M >> (23 - E)); // MÂ∑≤ÁªèË¢´Â∑¶Áßª‰∫Ü23‰Ωç
    if (s) V *= -1;  // Ë¥üÊï∞
    return V;
}
/* 
 * floatPower2 - Return bit-level equivalent of the expression 2.0^x
 *   (2.0 raised to the power x) for any 32-bit integer x.
 *
 *   The unsigned value that is returned should have the identical bit
 *   representation as the single-precision floating-point number 2.0^x.
 *   If the result is too small to be represented as a denorm, return
 *   0. If too large, return +INF.
 * 
 *   Legal ops: Any integer/unsigned operations incl. ||, &&. Also if, while 
 *   Max ops: 30 
 *   Rating: 4
 */
unsigned floatPower2(int x) {
    // V = (-1)^s * M * 2^E 
    // E = e-127  E‚àà[-126,127]
    // s=0,f=0, xÂ∏¶ÂÖ•EÂç≥ÂèØÂæóÂà∞ 2^x
    // ‰∏∫Ê≠§ÔºåÂèçÊºî‰ªéEÊ±ÇeÁöÑËøáÁ®ã
    if (x >= 128) return 0x7f800000;  // +INF
    if (x >= -126) return (x + 127) << 23; // [-126,127]: 2^x‰∏∫ËßÑÊ†ºÊï∞, f=1.0, e=x+127
    if (x >= -150) return 1 << (x + 150); // [-150,-125]: 2^x‰∏∫ÈùûËßÑÊ†ºÊï∞, f=2^(x+150), e=0
    else return 0; // too small
}

 *     and 0 otherwise 
 *   Legal ops: ! ~ & ^ | +
 *   Max ops: 10
 *   Rating: 1
 */
int isTmax(int x) {
  // ! ÈÄªËæëÂèñÂèçÔºå‰ªÖÂÖ®0ËøîÂõû1ÔºåÂÖ∂‰ΩôÊÉÖÂÜµÈÉΩËøîÂõû0
  // ~ Êåâ‰ΩçÂèñÂèç

  // ‰∏§‰∏™Áâπ‰æãÔºö~tmin+1=tmin; ~0+1=0. Áõ∏ÂèçÊï∞‰∏∫Ëá™Ë∫´ 
  // ÂæóÂà∞Âá†‰∏™ÂáΩÊï∞Ôºö
  // negate(x) (~x+1) // ÂèñÁõ∏ÂèçÊï∞
  // iszero(x) (!!x) // ‰ªÖÂΩì!!0=0
  // equal(x,y) !(x^y) // Âà§Êñ≠Áõ∏Á≠â

 *            isAsciiDigit(0x3a) = 0.

int isAsciiDigit(int x) {
  //    x > y
  // => x+(~y+1) > 0 // x+(-y)>0
  // => !(x+(~y+1) >> 31) // ‰ΩøÁî®Á¨¶Âè∑‰ΩçÂà§Êñ≠>0

 * logicalNeg - implement the ! operator, using all of

 *   Examples: logicalNeg(3) = 0, logicalNeg(0) = 1

int logicalNeg(int x) {

/* howManyBits - return the minimum number of bits required to represent x in
 *             two's complement
 *  Examples: howManyBits(12) = 5 // [-16,15]
 *            howManyBits(298) = 10 // [-512,511]
 *            howManyBits(-5) = 4 // [-8,7]
 *            howManyBits(0)  = 1
 *            howManyBits(-1) = 1 // [-1,0]
 *            howManyBits(0x80000000) = 32
 *  Legal ops: ! ~ & ^ | + << >>
 *  Max ops: 90
 *  Rating: 4
 */
int howManyBits(int x) {
  // Ë¥üÊï∞ÂèñÂèçÔºåÁªü‰∏ÄÂ§ÑÁêÜ
  // x>>31 ? ~x:x = x>>31 & ~x | ~(x>>31) & x

  // ‰ΩçÂÆΩÂèñÂÜ≥‰∫éÊúÄÈ´ò‰ΩçÔºåÈóÆÈ¢òËΩ¨Âåñ‰∏∫ÂØªÊâæÊúÄÈ´ò‰Ωç

  // !!(x>>16) Âà§Êñ≠xÁöÑÈ´ò16‰ΩçÊòØÂê¶Â≠òÂú®1
  int b16 = !!(x>>16) << 4;   
  // Ëã•Â≠òÂú®ÔºåËá≥Â∞ëÈúÄË¶Å16‰ΩçÔºåÂõ†Ê≠§b16ËµãÂÄº‰∏∫16Êàñ0
  x = x >> b16; 
  // ÈÄöËøáÁßª‰ΩçÂÆûÁé∞‰∫åÂàÜÊêúÁ¥¢Ôºö
  // Ëã•È´ò‰ΩçÂ≠òÂú®ÂàôËàçÂºÉ‰Ωé‰ΩçÔºåÈ´ò‰ΩçÂÖ®0ÂàôÂà§Êñ≠‰Ωé‰Ωç
  int b8 = !!(x>>8) << 3;
  x = x >> b8;

/* 
 * floatScale2 - Return bit-level equivalent of expression 2*f for
 *   floating point argument f.
 *   Both the argument and result are passed as unsigned int's, but
 *   they are to be interpreted as the bit-level representation of
 *   single-precision floating point values.
 *   When argument is NaN, return argument
 *   Legal ops: Any integer/unsigned operations incl. ||, &&. also if, while
 *   Max ops: 30
 *   Rating: 4
 */
unsigned floatScale2(unsigned uf) {
    unsigned s = uf & (1 << 31);
    unsigned exp = (uf & 0x7f800000) >> 23;
    unsigned frac = uf & (~0xff800000);

    if (exp == 0) return frac << 1 | s; // ÈùûËßÑÊ†ºÊï∞‰πò2Âç≥ÂèØ
    if (exp == 255) return uf;  //  ÁâπÊÆäÂÄºÁõ¥Êé•ËøîÂõû
    exp++;
    if (exp == 255) return 0x7f800000 | s;  // ‰πòÊ≥ï‰∏çÂèØËÉΩÂæóÂà∞NaNÔºåÊâÄ‰ª•ËøîÂõûÊó†Á©∑
    return s | (exp << 23) | frac;
}
/* 
 * floatFloat2Int - Return bit-level equivalent of expression (int) f
 *   for floating point argument f.
 *   Argument is passed as unsigned int, but
 *   it is to be interpreted as the bit-level representation of a
 *   single-precision floating point value.
 *   Anything out of range (including NaN and infinity) should return
 *   0x80000000u.
 *   Legal ops: Any integer/unsigned operations incl. ||, &&. also if, while
 *   Max ops: 30
 *   Rating: 4
 */
int floatFloat2Int(unsigned uf) {
    unsigned s = uf & (1 << 31);
    unsigned exp = (uf & 0x7f800000) >> 23;
    unsigned frac = uf & (~0xff800000);

    int E = exp - 127;
    if (exp == 255 || E > 31) return 0x80000000;  // ÁâπÊÆäÂÄº
    if (E < 0) return 0; // Â∞èÊï∞ËàçÂÖ•‰∏∫0

    unsigned M = frac | (1 << 23);  // 1 + frac
    int V = (E > 23 ? M << (E - 23) : M >> (23 - E)); // MÂ∑≤ÁªèË¢´Â∑¶Áßª‰∫Ü23‰Ωç
    if (s) V *= -1;  // Ë¥üÊï∞
    return V;
}
/* 
 * floatPower2 - Return bit-level equivalent of the expression 2.0^x
 *   (2.0 raised to the power x) for any 32-bit integer x.
 *
 *   The unsigned value that is returned should have the identical bit
 *   representation as the single-precision floating-point number 2.0^x.
 *   If the result is too small to be represented as a denorm, return
 *   0. If too large, return +INF.
 * 
 *   Legal ops: Any integer/unsigned operations incl. ||, &&. Also if, while 
 *   Max ops: 30 
 *   Rating: 4
 */
unsigned floatPower2(int x) {
    // V = (-1)^s * M * 2^E 
    // E = e-127  E‚àà[-126,127]
    // s=0,f=0, xÂ∏¶ÂÖ•EÂç≥ÂèØÂæóÂà∞ 2^x
    // ‰∏∫Ê≠§ÔºåÂèçÊºî‰ªéEÊ±ÇeÁöÑËøáÁ®ã
    if (x >= 128) return 0x7f800000;  // +INF
    if (x >= -126) return (x + 127) << 23; // [-126,127]: 2^x‰∏∫ËßÑÊ†ºÊï∞, f=1.0, e=x+127
    if (x >= -150) return 1 << (x + 150); // [-150,-125]: 2^x‰∏∫ÈùûËßÑÊ†ºÊï∞, f=2^(x+150), e=0
    else return 0; // too small
}
